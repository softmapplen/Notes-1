# 1. 字符串模式匹配KWP算法

获取部分匹配数组next

~~~java
    public static void getNext(String str,int[]next){
        int j=0;
        int k=-1;
        next[0]=-1;
        int len=str.length();
        while(j<len-1) {
            if (k == -1 || str.charAt(j) == str.charAt(k)) {
                ++j;
                ++k;
                next[j] = k;
            } else {
                k = next[k];
            }
        }
        System.out.print("next:[");
        for (int i:next)
            System.out.print(i+" ");
        System.out.println("]");
    }
~~~

~~~java
    public static int indexKMP(String s,String pattern){
       int i=0;
       int j=0;
       int slen=s.length();
       int plen=pattern.length();
       int []next=new int[plen];
       getNext(pattern,next);
       while(i<slen && j<plen){
           if(s.charAt(i)==pattern.charAt(j)){
               i++;
               j++;
           }else{
               if(next[j]==-1){
                   i++;
                   j=0;
               }else{
                   j=next[j];
               }
           }
           if(j==plen)
               return i-j;
       }
       return -1;
    }
~~~
